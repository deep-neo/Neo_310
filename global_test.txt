# Number of lines, words and character from file :-
def file_detail(file_name):

    with open(file_name, 'r') as file:
        line = 0; word = 0; char = 0

        for i in file:
            if sum(not str.isspace() for str in i):
                line += 1; word += len(i.split()); char += sum(not str.isspace() for str in i)

    print(f'# In the [31m{file_name} [30mfile \nâ€¢ number of lines : {line} | number of words : {word} | number of characters : {char}')

if __name__ == '__main__':
    file_detail('Automation.py')
if __name__ == '__main__':
    file_detail('Prime.Coloron.txt')
if __name__ == '__main__':
    file_detail('black.pearl.txt')
